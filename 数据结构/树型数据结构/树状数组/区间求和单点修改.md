```cpp
#include<iostream>
using namespace std;

int n,Q;
int tree[2000006];

inline int lowbit(int xx){return (xx&(-xx));}

inline void add(int pos,int val)
{
	while (pos<=n){
		tree[pos]+=val; pos+=lowbit(pos);
	}
}
inline int query(int pos)
{
	int val=0;
	while (pos){
		val+=tree[pos]; pos-=lowbit(pos);
	}
	return val;
}

int main()
{
	cin>>n>>Q;
	for (int i=1,x;i<=n;i++){
		cin>>x;
		add(i,x),add(i+1,-x);
	}
	for (int i=1,type,x,y,z;i<=Q;i++){
		cin>>type;
		if (type==1) {
			cin>>x>>y>>z;
			add(x,z),add(y+1,-z);
		}else {
			cin>>x;
			cout<<query(x)<<endl;
		}
	}
	
}
```
